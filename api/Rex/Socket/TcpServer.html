  self.create_param(Rex::Socket::Parameters.from_hash(hash))

def accept(opts = {})
  t = super()

  # jRuby compatibility
  if t.respond_to?('[]')
    t = t[0]
  end

  if (t)
    t.extend(Rex::Socket::Tcp)
    t.context = self.context

    pn = t.getpeername_as_array

    # We hit a "getpeername(2)" from Ruby
    return nil unless pn

    t.peerhost = pn[1]
    t.peerport = pn[2]
  end

  t
end
