def initialize_abstraction
  self.lsock, self.rsock = Rex::Socket.tcp_socket_pair()
  self.lsock.extend(Rex::IO::Stream)
  self.lsock.extend(Ext)
  self.rsock.extend(Rex::IO::Stream)

  self.monitor_rsock("StreamMonitorRemote")
end
