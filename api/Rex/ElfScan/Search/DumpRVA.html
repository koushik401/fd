def scan(param)
  config(param)

  $stdout.puts "[#{param['file']}]"

  # Adjust based on -A and -B flags
  pre = param['before'] || 0
  suf = param['after']  || 16

  @address -= pre
  @address = 0 if (@address < 0 || ! @address)
  buf = elf.read_rva(@address, suf)
  $stdout.puts elf.ptr_s(@address) + " " + buf.unpack("H*")[0]
end
